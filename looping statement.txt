    scanf("%d",&num);
    temp=num;
    while(num!=0){
        int num1=num%10;
         int t=1;
          fact=1;
         while(t<=num1){
            
            fact=fact*t;
            t++;
           }
           sum=sum+fact;
    
         num=num/10;
         }
        
         if(temp==sum)
                printf("it is strong number");
            else

}

// prime number

#include<stdio.h>
int main(){
    int num,count;
    scanf("%d",&num);
    for(int i=2;i<=num/2;i++){
        if(num%i==0)
            count++;
    }
    printf("%d\n",count);
    if(count==0)
        printf("it is prime number");
    else
        printf("it not prime number");


    return 0;
}

#include<stdio.h>
int main(){
    int n1,n2,sum1=1,sum2=1;
    scanf("%d%d",&n1,&n2);
    for(int i=2;i<=n1/2;i++)
    {
        if(n1%i==0)
            sum1=sum1+i;
    }
    printf("%d\n",sum1);
    for(int i=2;i<=n2/2;i++)
    {
        if(n2%i==0)
            sum2=sum2+i;
    }
    printf("%d\n",sum2);
    if(n1==sum2&&n2==sum1)
        printf("it is amicable number");
    return 0;
